1. Open a directory
    # open a directory (show all files and sub-dir)
    > open <directory_name>

2. Playlist 
    # List all playlists
    > playlist -l

    # View a playlist
    > playlist -v <playlist_name>

    # Create a playlist
    > playlist -c <playlist_name>

    # Update a playlist
    > playlist -u <playlist_name>

    # Delete a playlist 
    > playlist -d <playlist_name>

3. Metadata of file 
    # View Metadata
    > metadata -v <file_name>

    # Edit metadata
    > metadata -e --change-value
    > metadata -e --add-new-key

4. Play music 
    # Play media
    > play <file_name>

    # Play or pause current song
    > play
    > pause

    # Play next or previous song
    > next
    > prev

    # Turn on or off auto next feature
    > auto --on
    > auto --off

5. Change volume
    # Change volume, arithmetic like
    > volume ++
    > volume --
    > volume += <integer>
    > volume -= <integer>
